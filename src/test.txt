#include <stdio.h>
#include <unistd.h>
#include <sys/stat.h>

{

    char filename[] = "akh_client_main.c";
    struct stat buf;
    int result = stat(filename, &buf);
    off_t size = buf.st_size;

    printf("result %d\n", result);

    return 0;
}
